# Orion v4.9.3 Hotfix Plan - Complete Recovery

## Problem Analysis

### Root Cause: Circular Import Chain
**v4.9.2 Build Failed Silently:**
1. TypeScript compilation (`tsc`) **FAILED** on self-imports of `@orion-ds/react` in stories
2. Because tsc failed, Vite never ran (build process exits after tsc error)
3. `npm run build:styles` still ran, creating only `theme.css` (9.8 KB)
4. **Release script did not validate dist/ completeness** — published broken package anyway

**Root Cause File**: `packages/react/tsconfig.build.json`
- Includes `.stories.tsx` files in compilation
- Stories import from `@orion-ds/react` package (self-reference)
- During build, package not yet exists → tsc fails
- **Fix**: Exclude stories from `tsconfig.build.json`

### Secondary Issues
1. **No pre-publish validation** — Released without checking if exports exist
2. **Release script silent failures** — `turbo run build` doesn't bubble up tsc errors properly
3. **No dist/ completeness check** — Should validate all exports have corresponding files

## Pre-Publish Validation Script

**File**: `packages/react/scripts/validate-dist-completeness.js`

**Purpose**: Ensures all package.json exports have corresponding dist/ files

```javascript
#!/usr/bin/env node

const fs = require('fs');
const path = require('path');

const ROOT = path.resolve(__dirname, '..');
const PACKAGE_JSON_PATH = path.join(ROOT, 'package.json');
const DIST_PATH = path.join(ROOT, 'dist');

const colors = {
  reset: '\x1b[0m',
  red: '\x1b[31m',
  green: '\x1b[32m',
  yellow: '\x1b[33m',
  cyan: '\x1b[36m',
};

function log(msg, color = '') {
  console.log(`${color}${msg}${colors.reset}`);
}

function logError(msg) {
  log(`✗ ${msg}`, colors.red);
}

function logSuccess(msg) {
  log(`✓ ${msg}`, colors.green);
}

function logInfo(msg) {
  log(`→ ${msg}`, colors.cyan);
}

// 1. Read package.json
log('\n=== Validating dist/ Completeness ===\n', colors.cyan);

let packageJson;
try {
  packageJson = JSON.parse(fs.readFileSync(PACKAGE_JSON_PATH, 'utf8'));
} catch (e) {
  logError(`Cannot read package.json: ${e.message}`);
  process.exit(1);
}

logInfo(`Package: ${packageJson.name}@${packageJson.version}`);
logInfo(`Checking dist/ at: ${DIST_PATH}`);

// 2. Check dist/ exists
if (!fs.existsSync(DIST_PATH)) {
  logError(`dist/ directory does not exist`);
  process.exit(1);
}

// 3. Extract all expected files from package.json exports
const expectedFiles = new Set();
const exports = packageJson.exports || {};

for (const [exportPath, exportDef] of Object.entries(exports)) {
  if (exportPath === '.') {
    // Main export: index.mjs, index.cjs, index.d.ts
    expectedFiles.add('index.mjs');
    expectedFiles.add('index.cjs');
    expectedFiles.add('index.d.ts');
  } else if (typeof exportDef === 'string') {
    // Direct string path (e.g., "./dist/styles.css")
    const filePath = exportDef.replace(/^\.\//, '');
    expectedFiles.add(filePath);
  } else if (typeof exportDef === 'object') {
    // Conditional exports object
    if (exportDef.types) {
      const filePath = exportDef.types.replace(/^\.\/dist\//, '');
      expectedFiles.add(filePath);
    }
    if (exportDef.import) {
      const filePath = exportDef.import.replace(/^\.\/dist\//, '');
      expectedFiles.add(filePath);
    }
    if (exportDef.require) {
      const filePath = exportDef.require.replace(/^\.\/dist\//, '');
      expectedFiles.add(filePath);
    }
  }
}

// 4. Check all expected files exist
const missingFiles = [];
const foundFiles = [];

for (const file of expectedFiles) {
  const fullPath = path.join(DIST_PATH, file);

  if (fs.existsSync(fullPath)) {
    const stat = fs.statSync(fullPath);
    const sizeKB = (stat.size / 1024).toFixed(2);

    // Check for empty/suspiciously small files
    if (stat.size === 0) {
      logError(`${file} is 0 bytes (empty file)`);
      missingFiles.push(file);
    } else if (file.endsWith('.js') && stat.size < 100) {
      logError(`${file} is suspiciously small (${sizeKB} KB)`);
      missingFiles.push(file);
    } else {
      foundFiles.push(file);
      logSuccess(`${file} (${sizeKB} KB)`);
    }
  } else {
    logError(`${file} not found`);
    missingFiles.push(file);
  }
}

// 5. Additional checks
logInfo(`\nAdditional checks:`);

// Check for directories that should exist
const expectedDirs = ['tokens', 'sections', 'blocks', 'templates'];
for (const dir of expectedDirs) {
  const dirPath = path.join(DIST_PATH, dir);
  if (fs.existsSync(dirPath) && fs.statSync(dirPath).isDirectory()) {
    logSuccess(`${dir}/ directory exists`);
  } else {
    logError(`${dir}/ directory missing`);
    missingFiles.push(`${dir}/`);
  }
}

// Summary
log('\n=== Summary ===\n', colors.cyan);
log(`Found: ${foundFiles.length} files`);
log(`Missing: ${missingFiles.length} files`);

if (missingFiles.length > 0) {
  log(`\nMissing files:`, colors.red);
  for (const file of missingFiles) {
    log(`  - ${file}`);
  }
  log(`\nBuild appears incomplete. Aborting publish.\n`, colors.red);
  process.exit(1);
}

logSuccess(`\nAll exports validated successfully!`);
log('\nPackage is ready to publish.\n', colors.green);
process.exit(0);
```


## Test Suite for CI/CD

**File**: `packages/react/tests/dist-completeness.test.ts`

```typescript
import { describe, it, expect, beforeAll } from 'vitest';
import fs from 'fs';
import path from 'path';

describe('dist/ Completeness (Pre-Publish Validation)', () => {
  const ROOT = path.resolve(__dirname, '..');
  const DIST_PATH = path.join(ROOT, 'dist');
  const PACKAGE_JSON = JSON.parse(
    fs.readFileSync(path.join(ROOT, 'package.json'), 'utf8')
  );

  beforeAll(() => {
    if (!fs.existsSync(DIST_PATH)) {
      throw new Error('dist/ does not exist. Run: npm run build');
    }
  });

  it('dist/ directory exists', () => {
    expect(fs.existsSync(DIST_PATH)).toBe(true);
  });

  it('contains index exports (index.mjs, index.cjs, index.d.ts)', () => {
    expect(fs.existsSync(path.join(DIST_PATH, 'index.mjs'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'index.cjs'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'index.d.ts'))).toBe(true);
  });

  it('contains client exports (client.mjs, client.cjs, client.d.ts)', () => {
    expect(fs.existsSync(path.join(DIST_PATH, 'client.mjs'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'client.cjs'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'client.d.ts'))).toBe(true);
  });

  it('contains all named entry points', () => {
    const entryPoints = ['tokens', 'sections', 'blocks', 'templates'];
    for (const ep of entryPoints) {
      expect(fs.existsSync(path.join(DIST_PATH, ep, 'index.mjs'))).toBe(true);
      expect(fs.existsSync(path.join(DIST_PATH, ep, 'index.cjs'))).toBe(true);
      expect(fs.existsSync(path.join(DIST_PATH, ep, 'index.d.ts'))).toBe(true);
    }
  });

  it('contains all optional entry points (chart, calendar, editor, dnd, rich)', () => {
    const optionalEPs = ['chart', 'calendar', 'editor', 'dnd', 'rich'];
    for (const ep of optionalEPs) {
      expect(fs.existsSync(path.join(DIST_PATH, `${ep}.mjs`))).toBe(true);
      expect(fs.existsSync(path.join(DIST_PATH, `${ep}.cjs`))).toBe(true);
      expect(fs.existsSync(path.join(DIST_PATH, `${ep}.d.ts`))).toBe(true);
    }
  });

  it('contains CSS assets (styles.css, blocks.css, theme.css)', () => {
    expect(fs.existsSync(path.join(DIST_PATH, 'styles.css'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'blocks.css'))).toBe(true);
    expect(fs.existsSync(path.join(DIST_PATH, 'theme.css'))).toBe(true);
  });

  it('all exports in package.json have corresponding files', () => {
    const exports = PACKAGE_JSON.exports || {};
    const missingFiles = [];

    for (const [exportPath, exportDef] of Object.entries(exports)) {
      if (typeof exportDef === 'string') {
        const filePath = exportDef.replace(/^\.\/dist\//, '');
        if (!fs.existsSync(path.join(DIST_PATH, filePath))) {
          missingFiles.push(filePath);
        }
      } else if (typeof exportDef === 'object') {
        for (const [key, value] of Object.entries(exportDef)) {
          if (typeof value === 'string' && value.includes('dist')) {
            const filePath = value.replace(/^\.\/dist\//, '');
            if (!fs.existsSync(path.join(DIST_PATH, filePath))) {
              missingFiles.push(filePath);
            }
          }
        }
      }
    }

    expect(missingFiles).toEqual([]);
  });

  it('no empty or suspiciously small JS files', () => {
    const distFiles = getAllFiles(DIST_PATH);
    const suspiciousFiles = [];

    for (const file of distFiles) {
      if (file.endsWith('.mjs') || file.endsWith('.cjs')) {
        const stat = fs.statSync(file);
        if (stat.size === 0) {
          suspiciousFiles.push(`${path.relative(DIST_PATH, file)} (0 bytes)`);
        } else if (stat.size < 100) {
          suspiciousFiles.push(`${path.relative(DIST_PATH, file)} (${stat.size} bytes - too small)`);
        }
      }
    }

    expect(suspiciousFiles).toEqual([]);
  });

  function getAllFiles(dir: string): string[] {
    const files: string[] = [];
    const entries = fs.readdirSync(dir);

    for (const entry of entries) {
      const fullPath = path.join(dir, entry);
      const stat = fs.statSync(fullPath);

      if (stat.isDirectory()) {
        files.push(...getAllFiles(fullPath));
      } else {
        files.push(fullPath);
      }
    }

    return files;
  }
});
```
